588 public void chaseAlgorithm(MazeObject ghost) {
589 Field.Direction dir = ghost.getDirection();
590 //create list of available directions
591 List<Field.Direction> availableDirections = new CopyOnWriteArrayList<>();
592 for (Field.Direction d : Field.Direction.values()) {
593 if (ghost.canMove(d)) {
594 availableDirections.add(d);
595 }
596 }
597 // if you can't continue in the same direction, choose random direction, try not to go back, if the only option is to go back, go back
598 if (!availableDirections.contains(dir)) {
599 availableDirections.remove(dir.opposite(dir));
600 if (availableDirections.size() == 0) {
601 dir = dir.opposite(dir);
602 } else {
603 Random rand = new Random();
604 dir = availableDirections.get(rand.nextInt(availableDirections.size()));
605 }
606 }
607 else {
608 // if you can continue in the same direction, choose random direction with 30% chance, but try not to go back
609 availableDirections.remove(dir.opposite(dir));
610 Random rand = new Random();
611 if (rand.nextInt(10) < 3) {
612 dir = availableDirections.get(rand.nextInt(availableDirections.size()));
613 }
614 }
615 ghost.setDirection(dir);
616 }
